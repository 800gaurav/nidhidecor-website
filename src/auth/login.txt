import React, { useState } from 'react';
import { Eye, EyeOff, User, Lock, ArrowLeft, Shield } from 'lucide-react';

// Theme colors
export const colors = {
  theme1: "#0671FF",
  theme2: "#02D396"
};

const LoginPage = () => {
  const [formData, setFormData] = useState({
    userId: '',
    password: '',
  });
  const [showPassword, setShowPassword] = useState(false);
  const [loading, setLoading] = useState(false);
  const [captcha, setCaptcha] = useState('A7B2');
  const [captchaInput, setCaptchaInput] = useState('');

  const handleChange = (e) => {
    setFormData((prev) => ({ ...prev, [e.target.name]: e.target.value }));
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    setLoading(true);
    
    // Simulate login process
    setTimeout(() => {
      setLoading(false);
      alert('Login functionality would be implemented here');
    }, 1500);
  };

  const generateNewCaptcha = () => {
    const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';
    let newCaptcha = '';
    for (let i = 0; i < 4; i++) {
      newCaptcha += chars.charAt(Math.floor(Math.random() * chars.length));
    }
    setCaptcha(newCaptcha);
  };

  return (
    <div className="min-h-screen flex items-center justify-center bg-gradient-to-br from-gray-50 to-gray-100 p-4">
      <div className="w-full max-w-md">
        {/* Back button */}
        <button className="flex items-center text-sm text-gray-600 mb-6 hover:text-gray-800 transition-colors">
          <ArrowLeft size={16} className="mr-1" />
          Back to Home
        </button>

        {/* Login card */}
        <div className="bg-white rounded-2xl shadow-xl overflow-hidden">
          {/* Header with gradient */}
          <div 
            className="h-2 w-full"
            style={{ background: `linear-gradient(to right, ${colors.theme1}, ${colors.theme2})` }}
          ></div>
          
          <div className="px-8 py-8">
            <div className="text-center mb-8">
              <div className="flex justify-center mb-4">
                <div className="w-16 h-16 rounded-full flex items-center justify-center bg-blue-50">
                  <Shield size={32} style={{ color: colors.theme1 }} />
                </div>
              </div>
              <h1 className="text-3xl font-bold text-gray-800 mb-2">Welcome Back</h1>
              <p className="text-gray-600">Sign in to access your account</p>
            </div>

            <form onSubmit={handleSubmit}>
              {/* User ID Field */}
              <div className="mb-5">
                <label className="block text-sm font-medium text-gray-700 mb-2">
                  User ID
                </label>
                <div className="relative">
                  <div className="absolute inset-y-0 left-0 pl-3 flex items-center pointer-events-none">
                    <User size={18} className="text-gray-400" />
                  </div>
                  <input
                    name="userId"
                    type="text"
                    value={formData.userId}
                    onChange={handleChange}
                    placeholder="Enter your user ID"
                    className="w-full pl-10 pr-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent bg-white text-gray-900 transition-all"
                    disabled={loading}
                  />
                </div>
              </div>

              {/* Password Field */}
              <div className="mb-5">
                <label className="block text-sm font-medium text-gray-700 mb-2">
                  Password
                </label>
                <div className="relative">
                  <div className="absolute inset-y-0 left-0 pl-3 flex items-center pointer-events-none">
                    <Lock size={18} className="text-gray-400" />
                  </div>
                  <input
                    name="password"
                    type={showPassword ? "text" : "password"}
                    value={formData.password}
                    onChange={handleChange}
                    placeholder="Enter your password"
                    className="w-full pl-10 pr-10 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent bg-white text-gray-900 transition-all"
                    disabled={loading}
                  />
                  <button
                    type="button"
                    onClick={() => setShowPassword(!showPassword)}
                    className="absolute right-3 top-1/2 transform -translate-y-1/2 text-gray-400 hover:text-gray-600 transition p-1"
                    disabled={loading}
                  >
                    {showPassword ? <EyeOff size={18} /> : <Eye size={18} />}
                  </button>
                </div>
              </div>

              {/* Captcha Field */}
              <div className="mb-6">
                <label className="block text-sm font-medium text-gray-700 mb-2">
                  Enter Captcha
                </label>
                <div className="flex items-center space-x-3">
                  <div 
                    className="flex-1 px-4 py-2 bg-gray-100 rounded-lg text-center font-mono text-lg font-bold tracking-wider"
                    style={{ color: colors.theme1 }}
                  >
                    {captcha}
                  </div>
                  <button
                    type="button"
                    onClick={generateNewCaptcha}
                    className="px-3 py-2 text-sm rounded-lg font-medium"
                    style={{ backgroundColor: colors.theme2, color: 'white' }}
                  >
                    Refresh
                  </button>
                </div>
                <input
                  type="text"
                  value={captchaInput}
                  onChange={(e) => setCaptchaInput(e.target.value)}
                  placeholder="Type the captcha above"
                  className="w-full mt-3 px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent bg-white text-gray-900 transition-all"
                  disabled={loading}
                />
              </div>

              {/* Forgot Password Link */}
              <div className="text-right mb-6">
                <a href="#" className="text-sm font-medium" style={{ color: colors.theme1 }}>
                  Forgot Password?
                </a>
              </div>

              {/* Login Button */}
              <button
                type="submit"
                className="w-full py-3 px-4 rounded-lg text-white font-semibold transition-all shadow-md hover:shadow-lg focus:outline-none focus:ring-2 focus:ring-offset-2 disabled:opacity-75"
                style={{ 
                  background: `linear-gradient(to right, ${colors.theme1}, ${colors.theme2})`,
                  focusRingColor: colors.theme1
                }}
                disabled={loading}
              >
                {loading ? (
                  <div className="flex items-center justify-center">
                    <div className="w-5 h-5 border-t-2 border-r-2 border-white rounded-full animate-spin mr-2"></div>
                    Signing in...
                  </div>
                ) : (
                  "Sign In"
                )}
              </button>
            </form>

            {/* Sign up link */}
            <div className="text-center mt-6">
              <span className="text-sm text-gray-600">
                Don't have an account?{" "}
                <a href="#" className="font-medium" style={{ color: colors.theme1 }}>
                  Sign up now
                </a>
              </span>
            </div>
          </div>
        </div>

        {/* Feature highlights */}
        <div className="grid grid-cols-1 md:grid-cols-3 gap-4 mt-8">
          <div className="bg-white p-4 rounded-lg shadow text-center">
            <div className="w-10 h-10 rounded-full flex items-center justify-center mx-auto mb-2" style={{ backgroundColor: `${colors.theme1}15` }}>
              <Shield size={20} style={{ color: colors.theme1 }} />
            </div>
            <h3 className="font-medium text-gray-800">Secure Login</h3>
            <p className="text-sm text-gray-600 mt-1">Bank-level security</p>
          </div>
          <div className="bg-white p-4 rounded-lg shadow text-center">
            <div className="w-10 h-10 rounded-full flex items-center justify-center mx-auto mb-2" style={{ backgroundColor: `${colors.theme2}15` }}>
              <User size={20} style={{ color: colors.theme2 }} />
            </div>
            <h3 className="font-medium text-gray-800">Easy Access</h3>
            <p className="text-sm text-gray-600 mt-1">All your data in one place</p>
          </div>
          <div className="bg-white p-4 rounded-lg shadow text-center">
            <div className="w-10 h-10 rounded-full flex items-center justify-center mx-auto mb-2" style={{ backgroundColor: `${colors.theme1}15` }}>
              <Lock size={20} style={{ color: colors.theme1 }} />
            </div>
            <h3 className="font-medium text-gray-800">Privacy First</h3>
            <p className="text-sm text-gray-600 mt-1">Your data is always protected</p>
          </div>
        </div>
      </div>
    </div>
  );
};

export default LoginPage;
